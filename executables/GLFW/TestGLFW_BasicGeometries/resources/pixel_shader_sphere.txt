out vec4 output_color; // "output_color" replaces "gl_FragColor" because glBindFragDataLocation(...) has been called

in vec3 vs_position;

uniform vec4 color;

void main()
{

  float modif = 1.0;
  
  output_color =  color * (0.5 + 0.5 * modif);

  output_color.w = color.w; // keep alpha from incomming color
}
