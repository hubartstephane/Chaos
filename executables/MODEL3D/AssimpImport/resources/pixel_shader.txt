out vec4 output_color; // "output_color" replaces "gl_FragColor" because glBindFragDataLocation(...) has been called

in vec3 ps_position;
in vec3 ps_normal;

void main()
{
	vec3  color           = vec3(0.0, 0.0, 1.0);
	vec3  light_direction = normalize(vec3(-1.0, -1.0, -1.0));
	float light_contrib   = max(0, dot(ps_normal, -light_direction));
	
	output_color.rgb = color * (light_contrib * 0.5 + 0.5);
	output_color.a   = 1.0;
}
